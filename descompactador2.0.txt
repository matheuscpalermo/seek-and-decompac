#Script detects compact files (.tar.gz) in the current directory, and it's subdirectories, decompacts them and deletes empty directories #left behind. 

while [ ! -z "$(find . -name '*.tar.gz')"  ]; do
	find . -name "*.tar.gz" -exec tar xfvz "{}" \; -exec rm "{}" \;	
done

find . -type d -empty -exec rmdir "{}" \;
